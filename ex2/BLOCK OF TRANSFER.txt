BLOCK OF TRANSFER
                OUTPUT 2500AD
.ORG 2000H
DATA SEGMENT
SRC DB 11H,22H,33H,44H,55H
DEST DB 00H,00H,00H,00H,00H
COUNT DB 05H
DATA ENDS
CODE SEGMENT
        ASSUME CS:CODE,DS:DATA
        MOV AX,DATA
        MOV DS,AX
        MOV SI,OFFSET SRC
        MOV DI,OFFSET DEST
        MOV CX,COUNT
L1:     MOV AL,[SI]
        MOV [DI],AL
        INC SI
        INC DI
        LOOP L1
        HLT
CODE ENDS
END
