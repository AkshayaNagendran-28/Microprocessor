OUTPUT 2500AD

.ORG 2000H

DATA SEGMENT
LIST DB 01H,02H,04H,08H
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE, DS:DATA

START:
    MOV AX, DATA
    MOV DS, AX

    MOV AL, 80H
    MOV DX, FFE6H
    OUT DX, AL

MAIN:
    MOV SI, OFFSET LIST
    MOV CL, 04H   ; only 4 steps (90° sequence)

L1:
    MOV AL, [SI]
    MOV DX, FFE0H
    OUT DX, AL
    CALL DELAY

    INC SI
    DEC CL
    JNZ L1

    JMP MAIN      ; repeat again

DELAY:
    MOV BX, AAAAH
L2:
    DEC BX
    JNZ L2
    RET

    HLT

CODE ENDS
END START
