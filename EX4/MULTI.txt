MULTI:

        OUTPUT 2500AD
.ORG 2000H

DATA SEGMENT
LIST1 DB 06H,7CH,39H
LIST2 DB 66H,79H,6DH
COUNT DB 03H
DATA ENDS

CODE SEGMENT
        ASSUME CS:CODE,DS:DATA

START:
        MOV AX,DATA
        MOV DS,AX

        ; Initialize 8255 (Mode 0, Port A output)
        MOV AL,80H
        MOV DX,FFE6H
        OUT DX,AL

REPEAT:
        MOV SI,OFFSET LIST1
        MOV CL,COUNT

L2:
        MOV AL,[SI]
        MOV DX,FFE0H
        OUT DX,AL
        LEA DI,LIST2
        MOV CH,03H
L1:     MOV AL,[DI]
        MOV DX,FFE2H
        OUT DX,AL
        CALL DELAY
        INC DI
        DEC CH
        JNZ L1
        INC SI
        DEC CL
        JNZ L2
        JMP REPEAT
DELAY:
        MOV AH,15H
L3:
        MOV BX,AAAAH
L4:
        DEC BX
        JNZ L4
        DEC AH
        JNZ L3
        RET
CODE ENDS
END
